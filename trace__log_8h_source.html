00170         void logEvent(const char *category, const char *name,
00171                       TraceEvent::Type type,
00172                       size_t id, T argA, U argB) {
00173             if (!enabled) return;
00174 
00175             ChunkTenant &cs = (thread_chunk.sentinel) ? thread_chunk
00176                                                       : shared_chunk;
00177 
00178             if (!cs.sentinel->acquire()) {
00179                 resetChunk(cs);
00180                 return;
00181             }
00182             // State is busy
00183             if (!cs.chunk || cs.chunk->isFull()) {
00184                 replaceChunk(cs);
00185             }
00186             if (!cs.chunk) return;
00187 
00188             cs.chunk->addEvent() = TraceEvent(
00189                     category, name, type, id,
00190                     {{TraceArgument(argA), TraceArgument(argB)}},
00191                     {{TraceArgument::getType<T>(), TraceArgument::getType<U>()}});
00192             cs.sentinel->release();
00193         };
